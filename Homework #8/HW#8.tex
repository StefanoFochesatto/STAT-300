\begin{exercise}{Supplimental} A random number generator produces two successive values $X$ and $Y$ with the joint
  distribution, on the support $0< x,y <1$
  \begin{equation*}
    f(x,y) = x + y.
  \end{equation*}
  \begin{enumerate}
    \item What are the marginal distributions for $X$ and $Y$.\\
    
    \solution Consider integrating the joint probability distribution over the support on $x$ and $y$,
    \begin{equation*}
      f_x(x) = \int_0^1 x + y dy = x + \dfrac{y^2}{2} |_{0}^{1} = x + \dfrac{1}{2}
    \end{equation*}
    \begin{equation*}
      f_y(y) = \int_0^1 x + y dx = y + \dfrac{x^2}{2} |_{0}^{1} = y + \dfrac{1}{2}
    \end{equation*}
    \vspace{.25in}


    \item Are $X$ and $Y$ independent why or why not.\\
    
    \solution Note that,
    \begin{equation*}
      f(x,y) \neq f_x(x) f_y(y) = xy +\dfrac{x}{2}+ \dfrac{y}{2} + \dfrac{1}{4}
    \end{equation*}
    Therefore the variable are dependent. 

  \end{enumerate}











\end{exercise}
